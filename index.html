import pandas as pd import js def load_timetable(event): # 입력된 사용자 이름을 가져옴 username = js.document.querySelector("#username").value if not username: # 이름이 입력되지 않은 경우 경고 메시지 표시 js.alert("Please enter your name.") return # 사용자 이름에 맞는 CSV 파일 이름 생성 filename = f"{username}.csv" try: # CSV 파일을 읽어 데이터프레임으로 변환 df = pd.read_csv(f"/mnt/data/{filename}") # 시간표를 화면에 렌더링 render_timetable(df) except FileNotFoundError: # 파일이 없을 경우 경고 메시지 표시 js.alert(f"No timetable found for {username}.") def render_timetable(df): # 데이터프레임을 HTML 테이블로 변환 table_html = df.to_html(classes='timetable', index=False) # HTML 테이블을 페이지에 삽입 js.document.querySelector("#timetable").innerHTML = table_html # 버튼 클릭 시 load_timetable 함수 호출 js.document.querySelector("#loadButton").addEventListener("click", load_timetable)
